Solution#

def solution(n, words):
    answers = []

    prev = ""
    for i in range(len(words)):
        turn = (i // n) + 1
        player = (i % n) +1 

        curr = words[i]
        
      

        if prev != "" and prev[-1] != curr[0] or curr in answers:
            return [player, turn]
        else:
            answers.append(curr)
            prev = curr

    return [0, 0]
