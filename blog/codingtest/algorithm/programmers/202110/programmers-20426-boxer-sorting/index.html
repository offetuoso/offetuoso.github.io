def solution(weights, head2head):
    answer = []
    player = []

    for i in range(len(weights)):
        win_cnt = 0
        total_cnt = 0
        win_heavier_cnt = 0
        for j in range(len(head2head[i])):
            if head2head[i][j] != "N": total_cnt += 1
            if head2head[i][j] == "W":
                win_cnt += 1
                if weights[i] < weights[j]:
                    win_heavier_cnt += 1

        data = []
        try:
            rate = (win_cnt / total_cnt) * 1000000
        except:
            rate = 0
        data.append(rate)  # ìŠ¹ë¥ 
        data.append(win_heavier_cnt)  # ë¬´ê±°ìš´ ì‚¬ëŒì—ê²Œ ìŠ¹ìˆ˜
        data.append(weights[i])  # ë¬´ê²Œ
        data.append(i + 1)  # ë²ˆí˜¸
        player.append(data)

    for i in sorted(player, key=lambda x: (-x[0], -x[1], -x[2], x[3])):
        answer.append(i[3])

    return answer
